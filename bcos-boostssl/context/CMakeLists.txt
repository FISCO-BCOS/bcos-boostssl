set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -DSTATICLIB")

file(GLOB SRC_LIST "*.cpp")
file(GLOB HEADERS "*.h")

add_library(${BOOSTSSL_CONTEXT_TARGET} ${SRC_LIST} ${HEADERS})
target_link_libraries(${BOOSTSSL_CONTEXT_TARGET} PUBLIC Boost::system Boost::filesystem ${BOOSTSSL_UTILITIES_TARGET})
target_compile_options(${BOOSTSSL_CONTEXT_TARGET} PRIVATE -Wno-error -Wno-unused-variable)

# install ssl context target
install(
    TARGETS ${BOOSTSSL_CONTEXT_TARGET}
    EXPORT "${TARGETS_EXPORT_NAME}"
    LIBRARY DESTINATION "${CMAKE_INSTALL_LIBDIR}"
    ARCHIVE DESTINATION "${CMAKE_INSTALL_LIBDIR}"
    RUNTIME DESTINATION "${CMAKE_INSTALL_BINDIR}"
    INCLUDES DESTINATION "${CMAKE_INSTALL_INCLUDEDIR}"
)
